C'est une excellente idée. Gérer les avances de frais est un classique qui fausse complètement les budgets si l'application ne le prévoit pas.

Pour intégrer cela de manière fluide, j'ai ajouté un module "Saisir une opération" directement sur ton Tableau de bord (Dashboard).

Il te permet de choisir entre "➖ Dépense" et "➕ Remboursement".

  • Si tu saisis une dépense : le solde diminue et l'enveloppe se vide.
  • Si tu saisis un remboursement : le solde remonte et l'argent est recrédité dans l'enveloppe correspondante, te redonnant ton budget réel.

Voici le fichier index.html complet. J'y ai inclus une logique de démonstration pour que tu voies l'impact direct sur les jauges. Tu n'as plus qu'à le charger sur GitHub :

<script>
// --- DONNÉES DE DÉMONSTRATION ---
let appData = {
 total: 2450.00,
 envelopes: [
   { id: 'env1', name: 'Alimentation', budget: 400, spent: 250, color: 'var(--amber)' },
   { id: 'env2', name: 'Carburant', budget: 150, spent: 80, color: 'var(--purple)' },
   { id: 'env3', name: 'Loisirs (Avances)', budget: 200, spent: 220, color: 'var(--green)' } // Enveloppe dans le rouge
 ]
};

// --- LOGIQUE DU DASHBOARD & OPÉRATIONS ---
function renderDashboard() {
 // 1. Mise à jour du Solde Total
 document.getElementById('total-balance').textContent = appData.total.toFixed(2).replace('.', ',') + ' €';

 // 2. Mise à jour de la liste déroulante des enveloppes
 const select = document.getElementById('op-env');
 select.innerHTML = appData.envelopes.map(e => `<option value="${e.id}">${e.name}</option>`).join('');

 // 3. Mise à jour visuelle des enveloppes
 const list = document.getElementById('env-list');
 list.innerHTML = appData.envelopes.map(e => {
   const left = e.budget - e.spent; // Ce qu'il reste
   const pct = Math.min(100, Math.max(0, (e.spent / e.budget) * 100)); // Pourcentage consommé
   const isOver = left < 0; // Si on a dépassé
   const barColor = isOver ? 'var(--red)' : e.color;
   const textColor = isOver ? 'var(--red)' : 'var(--text)';

   return `
     <div class="env-card" style="${isOver ? 'border-color: rgba(248,113,113,0.3);' : ''}">
       <div style="display:flex; justify-content:space-between; margin-bottom:8px;">
         <span style="font-weight:600; font-size:15px;">${e.name}</span>
         <span style="color:var(--muted); font-size:12px">Budget: ${e.budget}€</span>
       </div>
       <div class="env-reste" style="color: ${textColor};">${left.toFixed(2).replace('.', ',')} €</div>
       <div style="font-size:11px; color:var(--muted); margin-top:-2px; margin-bottom:8px;">restant</div>
       <div class="env-bar-wrap"><div class="env-bar" style="width:${pct}%; background:${barColor}"></div></div>
     </div>
   `;
 }).join('');
}

// Fonction appelée quand on clique sur "Valider"
function submitOperation() {
 const type = document.getElementById('op-type').value; // 'expense' ou 'refund'
 const amountInput = document.getElementById('op-amount');
 const amount = parseFloat(amountInput.value);
 const envId = document.getElementById('op-env').value;

 if(isNaN(amount) || amount <= 0) return; // Sécurité si case vide

 const env = appData.envelopes.find(e => e.id === envId);
 if(!env) return;

 if(type === 'expense') {
   appData.total -= amount; // Retire du solde général
   env.spent += amount;     // Augmente les dépenses de l'enveloppe
 } else if(type === 'refund') {
   appData.total += amount; // Ajoute au solde général
   env.spent -= amount;     // DIMINUE les dépenses (Recrédite l'enveloppe)
 }

 // Reset le champ et met à jour l'affichage
 amountInput.value = '';
 renderDashboard();
}

// Initialisation au lancement
renderDashboard();


// --- NAVIGATION ONGLETS ---
function showPage(id) {
 document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
 document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
 document.getElementById('page-' + id).classList.add('active');
 event.currentTarget.classList.add('active');
}

// --- LOGIQUE CODE PIN & SPLASH ---
let lockInput = '';
const PIN = '123456';

function numPress(n) {
 if (lockInput.length < 6) {
   lockInput += n;
   updateDots();
   if (lockInput.length === 6) {
     setTimeout(() => {
       if (lockInput === PIN) {
         document.getElementById('lock').style.display = 'none';
       } else {
         lockInput = '';
         updateDots();
         document.getElementById('lock-error').textContent = 'Code incorrect';
       }
     }, 200);
   }
 }
}
function numDel() { lockInput = lockInput.slice(0, -1); updateDots(); }
function updateDots() {
 for (let i = 0; i < 6; i++) {
   document.getElementById('ld' + i).classList.toggle('filled', i < lockInput.length);
 }
}

   setTimeout(() => {
     document.getElementById('splash').style.display = 'none';
     document.getElementById('lock').style.display = 'flex';
   }, 500);
 }, 1500);
});
